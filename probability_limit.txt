// Code to simulate probability of getting sum of dots on faces of dice to be 3
clc;
clear all;
N = 100;
prob = [];
count = 0;

for i = 1:N
    // Simulate rolling two dice
    y1 = ceil(rand(1) * 6);
    y2 = ceil(rand(1) * 6);
    
    // Check if the sum is 3
    if (y1 + y2) == 3 then
        count = count + 1;
    end
    
    // Calculate running probability
    prob(i) = count / i;
end

disp(prob)
plot(prob)

xlabel('Number of trials');
ylabel('Probability');
title('Probability of getting sum of dots on faces of a dice to be 3');
