clc
m = input("Enter number of rows in first Matrix: ");
n = input("Enter number of columns in first Matrix: ");
p = input("Enter number of rows in Second Matrix: ");
q = input("Enter number of columns in Second Matrix: ");

if n == p then
    disp('Matrices are conformable for multiplication')
    
    // Input for first matrix
    disp('Enter the first Matrix')
    for i = 1:m
        for j = 1:n
            A(i,j) = input(" ");
        end
    end
    
    // Input for second matrix
    disp('Enter the second matrix')
    for i = 1:p
        for j = 1:q
            B(i,j) = input(" ");
        end
    end
    
    // Multiplication logic
    C = zeros(m, q);
    for i = 1:m
        for j = 1:q
            for k = 1:n
                C(i,j) = C(i,j) + A(i,k) * B(k,j);
            end
        end
    end
    
    // Display results
    disp('The first matrix is')
    disp(A)
    disp('The second matrix is')
    disp(B)
    disp('The product of two matrices is')
    disp(C)

else
    disp('Matrices are not conformable for multiplication')
end
