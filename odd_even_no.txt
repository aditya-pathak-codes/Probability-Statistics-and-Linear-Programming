clc;
disp('Name Roll No.') // Replace with your actual details
n = input('Enter the Number: ');

// Logic for Even Numbers
disp('Even Number');
for i = 1:n
    if modulo(i, 2) == 0 then
        disp(i);
    else
        continue;
    end
end

// Logic for Odd Numbers
disp('Odd numbers');
for i = 1:n
    if modulo(i, 2) ~= 0 then
        disp(i);
    else
        continue;
    end
end
